* Arguments and setting options

#+begin_src sh :results raw :exports results
echo '| Name | Description | Type | Default value |'
echo '|------+-------------+------+---------------|'
# Determine target name
source ./build_funcs.sh
TARGET=$(get_target)
# Check if target exists
if [ -f $TARGET ]; then
    # Just execute
    $TARGET --print-setting-entries
else
    # Build and then execute
    ./build.sh > /dev/null
    $TARGET --print-setting-entries
fi
#+end_src
